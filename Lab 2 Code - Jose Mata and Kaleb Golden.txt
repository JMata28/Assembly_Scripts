  ;Average Function 
  
  
   AVGFUNC:
   
  
  
  LDAA Numbers
  LDAB Numbers+1
  ABA

  ABA
  LDAB Numbers+3
  ABA
  LDAB Numbers+4
  ABA  
  LDAB Numbers+5

  ABA
  
  LSRA
  LSRA
  LSRA 
  
  STAA Avg
  
  
              LDAA #$00;Set port T as input
              STAA DDRT
  
  
      
              LDAA #$FF
              STAA DDRAD
  ReadInput:  LDAA PTIT
   
               CMPA #$00             ;If input is 00, do nothing.
               BEQ  DoNothing
              
  
               CMPA #%01             ;If input is 01, display the max value
               BEQ DisplayMax
  
               CMPA #%10             ;If input is 10, display the min value 
               BEQ DisplayMin
  
               CMPA #%11             ;If input is 11, display the average value
               BEQ DisplayAvg
  
  
  ;Labels to output values
  DisplayMin:
  
              LDAA Min
              STAA PTAD
              BRA ReadInput              
  
  DisplayMax:
  
              LDAA Max
              STAA PTAD
              BRA ReadInput
  
  DisplayAvg:
  
              LDAA Avg
              STAA PTAD
              BRA ReadInput
  
  DoNothing:
               LDAA #$00
               STAA PTAD
               BRA ReadInput
  
  Forever:
  LDAA #$FF
  STAA DDRAD
  LDAA #$00
  STAA PTAD
  LDAA PTIT
  BRA Forever
